We live in a 3D space $x,y,z$, the probability of finding a particle within a box $dx,dy,dz$
\begin{align}
dP = |\psi(x,y,z)|^{2}dxdydz
\end{align}
if $\psi(x,y,z) = \phi_{1}(x)\phi_{2}(y)\phi_{3}(z)$, then
\begin{align}
dP = |\phi_{1}(x)|^{2}dx\cdot |\phi_{2}(y)|^{2}dy\cdot |\phi_{3}(z)|^{2}dz
\end{align}

\begin{myboxes}{Example}{}
3-D harmonic oscillator. A particle in potential
$$V(x,y,z) = V_{x} + V_{y} + V_{z} = \frac{1}{2} m\omega^{2}(x^{2} + y^{2} + z^{2}),$$
$$H = \frac{p_{x}^{2}}{2m} + \frac{p_{y}^{2}}{2m} + \frac{p_{z}^{2}}{2m} + V_{x} + V_{y} + V_{z}$$
\end{myboxes}
